routine IsPalindrome(original: integer): boolean is
    var reversable: integer is 0;
    var temp: integer;
    temp := original;
    while temp > 0 loop 
        reversable := reversable * 10 + temp % 10;
        temp := temp/10;
    end; 
    if original=reversable then 
        return true;
    else 
        return false;
    end;
end;
