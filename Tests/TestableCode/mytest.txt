routine EntryPoint(): boolean is 
    var arr1: array [10] integer;
    var arr2: array [10] integer;
    
    for i in 1..10 loop
        arr1[i] := i;
    end;
    
    for i in reverse 1..10 loop
        arr2[i] := 1;
    end;
    
    var isEqual is true;
    for i in 1 .. 10 loop
        if arr1[i] /= arr2[i] then
            isEqual := false;
        end;   
    end;
    return isEqual;
end;
