var size is 10;

routine EntryPoint(): boolean is 
    var arr1: arr [size] integer;
    var arr2: arr [size] integer;
    for i in 1..size loop
        arr1[i] := i;
    end;
    for i in reverse 1..size loop
        arr2[i] := i;
    end;
    
    return checkEqual(arr1, arr2, size);
end;

routine BubbleSort(beforeArray: array [] integer, afterArray: array [] integer ) is
    
    var swapped: boolean;
    var temp: integer;
    
    for i in 1 .. size loop
        swapped := false;
        for j in 1 .. (size - i) loop
            if beforeArray[j] > beforeArray[j+1] then
                temp := beforeArray[j];
                beforeArray[j] := beforeArray[j + 1];
                beforeArray[j + 1] := temp;
                swapped := true;
            end;
        if swapped = false then
            return;
        end;
end;

routine checkEqual(beforeArray: array [] integer, afterArray: array [] integer): boolean is
    for i in size loop
        if beforeArray[i] /= afterArray[i] then
            return false;
        end;   
    end;
    return true;
end;
