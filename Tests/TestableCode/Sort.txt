routine EntryPoint(): boolean is 
    var arr1: array [10] integer;
    var arr2: array [10] integer;
    
    for i in 0..9 loop
        arr1[i] := i;
    end;
    
    for i in reverse 0..9 loop
        arr2[i] := i;
    end;
    
    BubbleSort(arr2);
    
    return CheckEqual(arr1, arr2);
end;

routine BubbleSort( beforeArray: array [] integer ) is
    
    var swapped: boolean;
    var temp: integer;
    
    for i in 0..9 loop
        swapped := false;
        for j in 0..(8 - i) loop
            if beforeArray[j] > beforeArray[j+1] then
                temp := beforeArray[j];
                beforeArray[j] := beforeArray[j + 1];
                beforeArray[j + 1] := temp;
                swapped := true;
            end;
        end;
        if swapped = false then
            return;
        end;
    end;
end;

routine CheckEqual(beforeArray: array [] integer, afterArray: array [] integer): boolean is
    for i in 0..9 loop
        if beforeArray[i] /= afterArray[i] then
            return false;
        end;   
    end;
    return true;
end;
